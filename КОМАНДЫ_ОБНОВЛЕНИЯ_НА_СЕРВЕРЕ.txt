========================================
УПРАВЛЕНИЕ БОТОМ НА СЕРВЕРЕ
========================================

# Перезапустить
sudo systemctl restart trading-bot.service

# Остановить
sudo systemctl stop trading-bot.service

# Запустить
sudo systemctl start trading-bot.service

# Статус
sudo systemctl status trading-bot.service --no-pager

# Логи в реальном времени
sudo journalctl -u trading-bot.service -f

# Последние 50 строк логов
sudo journalctl -u trading-bot.service -n 50 --no-pager

========================================
ПРОВЕРКА УСПЕШНОГО ОБНОВЛЕНИЯ
========================================

# Статус должен быть: Active: active (running)
sudo systemctl status trading-bot.service

# Проверить ошибки в логах
sudo journalctl -u trading-bot.service -n 50 | grep -i error
